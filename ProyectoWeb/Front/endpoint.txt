2. server.js - Endpoints completos

javascript
const express = require('express');
const cors = require('cors');
const bodyParser = require('body-parser');

const app = express();
const PORT = process.env.PORT || 3000;
const BACKEND_URL = process.env.BACKEND_URL || 'http://localhost:8000'; // Tu backend Python

// Middleware
app.use(cors({
    origin: '*',
    credentials: true,
    methods: ['GET', 'POST', 'PUT', 'DELETE', 'OPTIONS'],
    allowedHeaders: ['Content-Type', 'Authorization']
}));
app.use(bodyParser.json({ limit: '10mb' }));
app.use(bodyParser.urlencoded({ extended: true }));
app.use(express.static('public'));

// 🔄 Cache simple en memoria (opcional)
const cache = new Map();

// 📋 ENDPOINTS IDENTICOS AL BACKEND PYTHON

// GET /api/tables - Listar tablas
app.get('/api/tables', async (req, res) => {
    try {
        const response = await fetch(`${BACKEND_URL}/api/tables`);
        const data = await response.json();
        res.json(data);
    } catch (error) {
        res.status(500).json({ error: 'Error obteniendo tablas', details: error.message });
    }
});

// GET /api/table/:table_name/structure - Estructura para formularios
app.get('/api/table/:table_name/structure', async (req, res) => {
    const { table_name } = req.params;
    try {
        const response = await fetch(`${BACKEND_URL}/api/table/${table_name}/structure`);
        if (!response.ok) {
            throw new Error(`HTTP ${response.status}: ${response.statusText}`);
        }
        const data = await response.json();
        res.json(data);
    } catch (error) {
        res.status(500).json({ error: `Error obteniendo estructura de ${table_name}`, details: error.message });
    }
});

// GET /api/table/:table_name - Lista paginada
app.get('/api/table/:table_name', async (req, res) => {
    const { table_name } = req.params;
    const { page = 1, limit = 20 } = req.query;
    
    try {
        const response = await fetch(
            `${BACKEND_URL}/api/table/${table_name}?page=${page}&limit=${limit}`
        );
        if (!response.ok) {
            throw new Error(`HTTP ${response.status}: ${response.statusText}`);
        }
        const data = await response.json();
        res.json(data);
    } catch (error) {
        res.status(500).json({ error: `Error obteniendo registros de ${table_name}`, details: error.message });
    }
});

// GET /api/table/:table_name/:record_id - Registro específico
app.get('/api/table/:table_name/:record_id', async (req, res) => {
    const { table_name, record_id } = req.params;
    try {
        const response = await fetch(`${BACKEND_URL}/api/table/${table_name}/${record_id}`);
        if (!response.ok) {
            const errorText = await response.text();
            return res.status(response.status).json({ 
                error: `Registro no encontrado`, 
                details: errorText 
            });
        }
        const data = await response.json();
        res.json(data);
    } catch (error) {
        res.status(500).json({ error: 'Error obteniendo registro', details: error.message });
    }
});

// POST /api/table/:table_name/insert - Crear registro
app.post('/api/table/:table_name/insert', async (req, res) => {
    const { table_name } = req.params;
    const data = req.body;
    
    try {
        const response = await fetch(`${BACKEND_URL}/api/table/${table_name}/insert`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify(data)
        });
        
        if (!response.ok) {
            const errorText = await response.text();
            throw new Error(errorText);
        }
        
        const result = await response.json();
        res.json(result);
    } catch (error) {
        res.status(400).json({ 
            error: 'Error creando registro', 
            details: error.message,
            table: table_name 
        });
    }
});

// PUT /api/table/:table_name/:record_id - Actualizar registro
app.put('/api/table/:table_name/:record_id', async (req, res) => {
    const { table_name, record_id } = req.params;
    const data = req.body;
    
    try {
        const response = await fetch(`${BACKEND_URL}/api/table/${table_name}/${record_id}`, {
            method: 'PUT',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify(data)
        });
        
        if (!response.ok) {
            const errorText = await response.text();
            throw new Error(errorText);
        }
        
        const result = await response.json();
        res.json(result);
    } catch (error) {
        res.status(400).json({ 
            error: 'Error actualizando registro', 
            details: error.message 
        });
    }
});

// DELETE /api/table/:table_name/:record_id - Eliminar registro
app.delete('/api/table/:table_name/:record_id', async (req, res) => {
    const { table_name, record_id } = req.params;
    
    try {
        const response = await fetch(`${BACKEND_URL}/api/table/${table_name}/${record_id}`, {
            method: 'DELETE'
        });
        
        if (!response.ok) {
            const errorText = await response.text();
            throw new Error(errorText);
        }
        
        const result = await response.json();
        res.json(result);
    } catch (error) {
        res.status(404).json({ 
            error: 'Error eliminando registro', 
            details: error.message 
        });
    }
});

// 🏠 Ruta raíz
app.get('/', (req, res) => {
    res.json({ 
        message: '🚀 Node.js DB CRUD API funcionando',
        backend: BACKEND_URL,
        endpoints: [
            'GET /api/tables',
            'GET /api/table/:table/structure',
            'GET /api/table/:table?page=1&limit=20',
            'POST /api/table/:table/insert',
            'PUT /api/table/:table/:id',
            'DELETE /api/table/:table/:id'
        ]
    });
});

// 404
app.use('*', (req, res) => {
    res.status(404).json({ error: 'Endpoint no encontrado' });
});

// 🚀 Iniciar servidor
app.listen(PORT, () => {
    console.log(`🎉 Servidor Node.js corriendo en http://localhost:${PORT}`);
    console.log(`🔗 Backend Python: ${BACKEND_URL}`);
    console.log(`📱 Frontend: http://localhost:${PORT}/form.html`);
});

3. 🚀 Instalación y ejecución

bash
# 1. Crear proyecto
mkdir db-crud-nodejs
cd db-crud-nodejs

# 2. Instalar dependencias
npm install

# 3. Copiar tu form.html a la carpeta public/
mkdir public
# Copia tu form.html aquí

# 4. Asegurar que tu backend Python esté corriendo en puerto 8000
# python run.py (en la carpeta del backend)

# 5. Iniciar Node.js
npm run dev  # Con hot reload
# o
npm start    # Producción

4. 🔧 Configuración Frontend (form.html)

Tu form.html debe apuntar al Node.js proxy en lugar del backend directo:

javascript
// ❌ ANTES (directo al Python)
const API_BASE = 'http://localhost:8000/api';

// ✅ DESPUÉS (a través de Node.js proxy)
const API_BASE = 'http://localhost:3000/api';  // Puerto de Node.js
